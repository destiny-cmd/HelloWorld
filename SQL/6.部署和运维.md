## 生产环境配置
在配置文件中配置生成环境，主要包括基础配置、连接配置、缓冲区配置、日志配置、引擎配置、安全配置。

my.ini

```sql
[client]
port = 3306
default-character-set = utf8mb4

[mysql]
default-character-set = utf8mb4
prompt = '\\u@\\h [\\d]> '

[mysqld]
# ==================== 基础配置 ====================
user = mysql
port = 3306
basedir = /usr/local/mysql
datadir = /var/lib/mysql
socket = /var/lib/mysql/mysql.sock
pid-file = /var/run/mysqld/mysqld.pid
character-set-server = utf8mb4
collation-server = utf8mb4_unicode_ci
init-connect = 'SET NAMES utf8mb4'

# ==================== 连接配置 ====================
max_connections = 200                  # 最大连接数，根据并发量调整
max_connect_errors = 100               # 最大连接错误数，防止暴力破解
connect_timeout = 10                   # 连接超时时间（秒）
wait_timeout = 28800                   # 空闲连接超时（秒），8小时
interactive_timeout = 28800            # 交互式连接超时

# ==================== 缓冲区配置 ====================
innodb_buffer_pool_size = 2G           # 建议设置为物理内存的70%-80%
innodb_log_buffer_size = 64M           # 日志缓冲区大小
innodb_buffer_pool_instances = 4       # 缓冲池实例数（CPU核心数）

key_buffer_size = 256M                 # MyISAM索引缓冲区（如仍有使用）
sort_buffer_size = 2M                  # 排序缓冲区
read_buffer_size = 2M                  # 顺序读缓冲区
read_rnd_buffer_size = 4M              # 随机读缓冲区
join_buffer_size = 2M                  # 连接缓冲区

tmp_table_size = 64M                   # 内存临时表大小
max_heap_table_size = 64M              # 内存表最大大小

# ==================== 日志配置 ====================
# 错误日志
log_error = /var/log/mysql/error.log
log_warnings = 2

# 通用查询日志（生产环境建议关闭）
# general_log = 0
# general_log_file = /var/log/mysql/mysql.log

# 慢查询日志
slow_query_log = ON
slow_query_log_file = /var/log/mysql/mysql-slow.log
long_query_time = 1                    # 超过1秒视为慢查询
log_queries_not_using_indexes = ON    # 记录未使用索引的查询
log_slow_admin_statements = ON         # 记录慢管理语句
min_examined_row_limit = 100           # 扫描行数阈值

# binlog配置（重要！用于主从复制和恢复）
log_bin = /var/lib/mysql/mysql-bin
log_bin_index = /var/lib/mysql/mysql-bin.index
binlog_format = ROW                    # 推荐ROW格式，更安全
binlog_row_image = FULL                # 记录完整行数据
binlog_cache_size = 32K
max_binlog_cache_size = 512M
max_binlog_size = 512M                 # 单个binlog文件大小
expire_logs_days = 7                   # binlog保留天数
sync_binlog = 1                        # 每次事务提交都同步binlog（最安全）

# ==================== InnoDB配置 ====================
innodb_data_file_path = ibdata1:100M:autoextend
innodb_file_per_table = ON             # 每张表独立表空间
innodb_autoinc_lock_mode = 2           # 自增锁模式（并发插入性能）
innodb_flush_log_at_trx_commit = 1     # 每次事务提交都刷盘（最安全）
innodb_flush_method = O_DIRECT         # 绕过操作系统缓存，直接读写磁盘
innodb_io_capacity = 2000              # IO能力（SSD可设更高）
innodb_io_capacity_max = 4000
innodb_lock_wait_timeout = 50          # 锁等待超时（秒）
innodb_deadlock_detect = ON            # 死锁检测开启

# ==================== 安全配置 ====================
# 禁止使用符号链接
symbolic-links = 0

# SQL模式（严格模式）
sql_mode = 'STRICT_TRANS_TABLES,NO_ZERO_IN_DATE,NO_ZERO_DATE,ERROR_FOR_DIVISION_BY_ZERO,NO_ENGINE_SUBSTITUTION'

# 默认存储引擎
default_storage_engine = InnoDB

# 限制导入导出路径
secure_file_priv = /var/lib/mysql-files

# 禁止本地文件读取
local-infile = 0

```



## 权限管理与用户创建
根据权限创建用户，包括应用用户、只读用户、备份用户、管理员用户。

```sql
-- ============================================================
-- 创建专用用户（最小权限原则）
-- ============================================================

-- 应用用户（拥有应用程序所需的全部权限）
CREATE USER 'app_user'@'192.168.1.%' IDENTIFIED BY 'StrongPassword2026!@#';
GRANT SELECT, INSERT, UPDATE, DELETE ON meeting_room.* TO 'app_user'@'192.168.1.%';

-- 只读用户（用于报表查询）
CREATE USER 'readonly_user'@'192.168.1.%' IDENTIFIED BY 'ReadOnly@2026';
GRANT SELECT ON meeting_room.* TO 'readonly_user'@'192.168.1.%';

-- 备份用户（用于mysqldump备份）
CREATE USER 'backup_user'@'localhost' IDENTIFIED BY 'BackupSecure@2026';
GRANT SELECT, LOCK TABLES, SHOW VIEW, PROCESS, REPLICATION CLIENT ON *.* TO 'backup_user'@'localhost';
GRANT SELECT ON performance_schema.* TO 'backup_user'@'localhost';

-- 管理员用户（仅限本地）
CREATE USER 'dba_admin'@'localhost' IDENTIFIED BY 'AdminSuper@2026';
GRANT ALL PRIVILEGES ON meeting_room.* TO 'dba_admin'@'localhost' WITH GRANT OPTION;

-- 刷新权限
FLUSH PRIVILEGES;

-- ============================================================
-- 查看用户权限
-- ============================================================
SHOW GRANTS FOR 'app_user'@'192.168.1.%';

```



## 数据备份
增量备份

```sql
#!/bin/bash
# ============================================================
# 增量备份脚本（基于binlog）
# ============================================================

BACKUP_DIR="/data/backup/mysql/binlog"
mkdir -p "${BACKUP_DIR}"

# 记录当前binlog位置
mysql -u backup_user -pBackupSecure@2026 -e "FLUSH LOGS;"

# 复制binlog文件
cp /var/lib/mysql/mysql-bin.* "${BACKUP_DIR}/"

# 清理备份目录中的旧binlog（保留最近3天）
find "${BACKUP_DIR}" -name "mysql-bin.*" -mtime +3 -delete

```

```sql

read_only = 1                     # 从库只读
```



主从复制配置

```sql
-- ===== 在主库执行 =====
-- 创建复制用户
CREATE USER 'replication'@'%' IDENTIFIED BY 'ReplSecure@2026';
GRANT REPLICATION SLAVE, REPLICATION CLIENT ON *.* TO 'replication'@'%';
FLUSH PRIVILEGES;

-- 查看主库状态（记录File和Position值）
SHOW MASTER STATUS;
-- File: mysql-bin.000001, Position: 154

-- ===== 在从库执行 =====
-- 配置主库连接
CHANGE MASTER TO
    MASTER_HOST = '192.168.1.10',
    MASTER_USER = 'replication',
    MASTER_PASSWORD = 'ReplSecure@2026',
    MASTER_LOG_FILE = 'mysql-bin.000001',
    MASTER_LOG_POS = 154;

-- 启动复制
START SLAVE;

-- 查看复制状态
SHOW SLAVE STATUS\G;
-- 检查: Slave_IO_Running = Yes, Slave_SQL_Running = Yes

```

## 性能调优
| **问题现象** | **可能原因** | **解决方案** |
| --- | --- | --- |
| 查询缓慢 | 缺少索引 | 使用EXPLAIN分析，添加合适的索引 |
| 连接数不足 | max_connections设置过小 | 调整max_connections参数 |
| 锁等待超时 | 长事务未提交 | 检查事务，优化业务逻辑，缩短事务时间 |
| 缓冲池命中率低 | innodb_buffer_pool_size太小 | 增加缓冲池大小（物理内存的70%-80%） |
| IO瓶颈 | 磁盘性能不足 | 更换SSD，调整innodb_io_capacity |

## 数据库运维手册
### 日常巡检命令
```sql
-- 快速健康检查
SHOW STATUS LIKE 'Uptime';               -- 运行时长
SHOW STATUS LIKE 'Threads_connected';    -- 当前连接数
SHOW STATUS LIKE 'Slow_queries';         -- 慢查询累计
SHOW PROCESSLIST;                        -- 当前执行的SQL

-- 查看数据库大小
SELECT 
    table_schema AS '数据库',
    ROUND(SUM(data_length + index_length) / 1024 / 1024, 2) AS '大小(MB)'
FROM information_schema.tables
WHERE table_schema = 'meeting_room';

-- 查看各表记录数
SELECT 
    table_name,
    table_rows,
    ROUND(data_length / 1024 / 1024, 2) AS data_mb
FROM information_schema.tables
WHERE table_schema = 'meeting_room'
ORDER BY table_rows DESC;
```



### 应急处理流程
数据库连接数耗尽处理

```sql
- 查看当前连接
SHOW PROCESSLIST;

-- 杀死长时间运行的查询（需谨慎）
KILL CONNECTION_ID;

-- 临时增加最大连接数
SET GLOBAL max_connections = 300;
```



误操作数据恢复处理

```sql
# 从最近的全量备份恢复
/usr/local/scripts/restore.sh /data/backup/mysql/meeting_room_full_20260622_020000.sql.gz

# 然后使用binlog恢复到误操作前的时间点
mysqlbinlog --stop-datetime="2026-06-22 10:30:00" /var/lib/mysql/mysql-bin.000001 | mysql -u dba_admin -p

```



主从复制中断处理

```sql
-- 查看从库状态
SHOW SLAVE STATUS\G

-- 如果出现错误，尝试跳过（需确认错误可忽略）
STOP SLAVE;
SET GLOBAL SQL_SLAVE_SKIP_COUNTER = 1;
START SLAVE;

-- 如果无法恢复，重新同步
STOP SLAVE;
RESET SLAVE;
CHANGE MASTER TO ...;
START SLAVE;
```
